Create table produtos(
idProduto integer not null auto_increment,
nomeProduto varchar(30) not null,
descriProduto text not null,
imagemProduto varchar(60) not null,
precoProduto decimal(10,2) not null,
estoqueProduto integer not null,
idCategoria integer,
CONSTRAINT fk_CateProd FOREIGN KEY (idCategoria) REFERENCES categorias (idCategoria),
primary key(idProduto)

);
Create table categorias(
    idCategoria integer not null auto_increment,
    nomeCategoria varchar(40) not null,
    imagemCategoria blob,
    descriCategoria text null,
    primary key(idCategoria)

);
Create table imagens(
    idImagem integer not null auto_increment,
    idProduto integer,
    imagemSecundaria blob not null,
    CONSTRAINT fk_ImgsProd FOREIGN KEY (idProduto) REFERENCES produtos (idProduto),
    primary key(idImagem)

);

Create table usuario(
    idUsuario intenger not null auto_increment,
    nomeUsuario varchar(16) not null,
    senhaUsuario varbinary(100) not null,
    primary key(idUsuario)

);
